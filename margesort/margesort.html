            Merge(left, right):
                merged = []
                while left is not empty and right is not empty:
                    if left[0] <= right[0]:
                        merged.append(left[0])
                        left = left[1:]
                    else:
                        merged.append(right[0])
                        right = right[1:]
            
                return merged + left + right
